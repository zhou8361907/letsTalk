### 1. ConversationContext（对话上下文）

**文件**: `src/ai_requirement_os/agents/conversation_context.py`

**职责**:
- 管理对话历史（不随页面切换清空）
- 记录访问过的页面
- 按需加载 Skills
- 对话压缩（使用 LLM 总结）

**核心方法**:
```python
class ConversationContext:
    """对话上下文（与页面解耦）"""
    
    def __init__(self):
        self.current_page_path: Optional[str] = None
        self.visited_pages: List[str] = []
        self.conversation_history: List[Message] = []
        self.loaded_skills_cache: Dict[str, List[Skill]] = {}
        self.max_history = 100
    
    def visit_page(self, page_path: str):
        """访问页面（不清空历史）"""
        self.current_page_path = page_path
        if page_path not in self.visited_pages:
            self.visited_pages.append(page_path)
    
    def add_message(self, role: str, content: str):
        """添加消息"""
        self.conversation_history.append(Message(role, content))
        
        # 自动压缩
        if len(self.conversation_history) > self.max_history:
            await self.compress_history()
    
    async def compress_history(self):
        """压缩对话历史（使用 LLM 总结）"""
        # 保留最近 20 条
        recent = self.conversation_history[-20:]
        
        # 总结前面的对话
        old_messages = self.conversation_history[:-20]
        summary = await summarize_conversation(old_messages)
        
        # 替换
        self.conversation_history = [
            Message("system", f"对话历史总结：{summary}")
        ] + recent
    
    def load_skills(self, page_path: str) -> List[Skill]:
        """按需加载 Skills"""
        if page_path not in self.loaded_skills_cache:
            skills = SkillDatabase.load_skills(page_path)
            self.loaded_skills_cache[page_path] = skills
        return self.loaded_skills_cache[page_path]
    
    def get_context_summary(self) -> str:
        """获取上下文摘要（用于 LLM）"""
        return f"""
当前页面: {self.current_page_path}
访问过的页面: {', '.join(self.visited_pages)}
对话轮次: {len(self.conversation_history)}

最近 5 条对话:
{format_recent_messages(self.conversation_history[-5:])}
"""
```

**skills_index.json 格式**:
```json

  "last_updated": "2026-05-21T17:00:00Z",
  "pages": {
    "Detail.vue": {
```

**核心方法**:
```python
class SkillDatabase:
    """Skills 数据库"""
    
    def __init__(self, base_path: str = ".agent/skills"):
        self.base_path = Path(base_path)
        self.index_file = self.base_path / "skills_index.json"
        self.index = self._load_index()
    
    def save_skills(self, page_path: str, skills: List[Skill]):
        """保存页面的 Skills"""
        # 保存到文件
        skill_file = self.base_path / f"{page_path}.json"
        skill_file.parent.mkdir(parents=True, exist_ok=True)
        
        with open(skill_file, 'w') as f:
            json.dump([skill.to_dict() for skill in skills], f, ensure_ascii=False, indent=2)
        
        # 更新索引
        self._update_index(page_path, skills)
    
    def load_skills(self, page_path: str) -> List[Skill]:
        """加载页面的 Skills"""
        skill_file = self.base_path / f"{page_path}.json"
        
        if not skill_file.exists():
            return []
        
        with open(skill_file, 'r') as f:
            data = json.load(f)
        
        return [Skill.from_dict(d) for d in data]
    
    def search_globally(
        self,
        keyword: str,
        skill_type: Optional[str] = None
    ) -> List[Tuple[str, Skill]]:
        """全局搜索 Skills"""
        results = []
        
        # 从索引中快速查找
        skill_ids = self.index.get("skill_index", {}).get(keyword, [])
        
        for skill_id in skill_ids:
            # 找到 Skill 所在的页面
            for page_path, page_info in self.index["pages"].items():
                for skill_info in page_info["skills"]:
                    if skill_info["id"] == skill_id:
                        # 加载完整的 Skill
                        skills = self.load_skills(page_path)
                        skill = next((s for s in skills if s.skill_id == skill_id), None)
                        if skill:
                            if skill_type is None or skill.skill_type == skill_type:
                                results.append((page_path, skill))
        
        return results
    
    def check_analyzed(self, page_path: str) -> bool:
        """检查页面是否已分析"""
        return page_path in self.index.get("pages", {})
    
    def get_analysis_time(self, page_path: str) -> Optional[str]:
        """获取分析时间"""
        page_info = self.index.get("pages", {}).get(page_path)
        return page_info.get("analyzed_at") if page_info else None
```

```python
class ConversationAgent:
    """对话 Agent（解耦页面）"""
    
    def __init__(self, llm_client, tools: List[Tool]):
        self.client = llm_client
        self.tools = {tool.name: tool for tool in tools}
        self.tool_schemas = [tool.to_openai_format() for tool in tools]
        self.context = ConversationContext()
        self.skill_db = SkillDatabase()
    
    def visit_page(self, page_path: str) -> Iterator[StreamEvent]:
        """访问页面（不清空历史）"""
        # 更新当前页面
        self.context.visit_page(page_path)
        
        yield StreamEvent("page_changed", {
            "page": page_path,
            "visited_pages": self.context.visited_pages
        })
        
        # 检查是否已分析
        if not self.skill_db.check_analyzed(page_path):
            yield StreamEvent("need_analysis", {
                "page": page_path,
                "estimated_time": "30-60秒"
            })
        else:
            # 加载 Skills
            skills = self.context.load_skills(page_path)
            yield StreamEvent("skills_loaded", {
                "page": page_path,
                "skills": [s.to_dict() for s in skills]
            })
    
    async def chat(self, user_message: str) -> Iterator[StreamEvent]:
        """对话（支持跨页面）"""
        # 添加消息
        self.context.add_message("user", user_message)
        
        # 构建系统提示
        system_prompt = self._build_system_prompt()
        
        # 构建消息
        messages = [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_message}
        ]
        
        # 运行 Agent 循环
        async for event in self._run_with_streaming(messages):
            yield event
    
    def _build_system_prompt(self) -> str:
        """构建系统提示"""
        context_summary = self.context.get_context_summary()
        
        return f"""你是一个专业的代码分析助手。

{context_summary}

## 可用工具

你可以使用以下工具：
- search_skills_globally: 在所有页面中搜索 Skills
- get_page_skills: 获取指定页面的 Skills
- get_skill_details: 获取 Skill 的详细信息
- find_similar_skills: 查找相似的 Skills
- analyze_page_async: 异步分析页面（如果未分析）

## 回答要求

1. 支持跨页面查询 - 用户可能问其他页面的功能
2. 主动搜索 - 如果当前页面没有，搜索其他页面
3. 详细回答 - 提供完整的实现路径和代码位置
4. 代码示例 - 展示关键代码片段

"""
```

### 4. 后台分析任务

**文件**: `src/ai_requirement_os/agents/analysis_task.py`

**职责**:
- 异步分析页面
- 推送分析进度
- 保存分析结果

**核心方法**:
```python
class AnalysisTask:
    """后台分析任务"""
    
    def __init__(self, page_path: str, task_id: str):
        self.page_path = page_path
        self.task_id = task_id
        self.progress = 0
        self.step = "准备分析"
        self.status = "pending"
        self.result = None
        self.error = None
    
    async def run(self):
        """运行分析任务"""
        try:
            self.status = "running"
            
            # 1. 解析 Vue 文件
            self.update_progress(20, "解析 Vue 文件")
            vue_ast = await parse_vue_file(self.page_path)
            
            # 2. 提取 API 调用
            self.update_progress(40, "提取 API 调用")
            api_calls = await extract_api_calls(vue_ast)
            
            # 3. 分析后端
            self.update_progress(60, "分析后端代码")
            backend_info = await analyze_backend(api_calls)
            
            # 4. 生成 Skills
            self.update_progress(80, "生成 Skills")
            skills = await generate_skills(vue_ast, api_calls, backend_info)
            
            # 5. 保存结果
            self.update_progress(95, "保存结果")
            SkillDatabase().save_skills(self.page_path, skills)
            
            # 完成
            self.update_progress(100, "完成")
            self.status = "completed"
            self.result = skills
            
        except Exception as e:
            self.status = "failed"
            self.error = str(e)
    
    def update_progress(self, progress: int, step: str):
        """更新进度"""
        self.progress = progress
        self.step = step

# 全局任务管理器
analysis_tasks: Dict[str, AnalysisTask] = {}

async def start_analysis(page_path: str) -> str:
    """启动分析任务"""
    task_id = generate_task_id()
    task = AnalysisTask(page_path, task_id)
    analysis_tasks[task_id] = task
    
    # 启动后台任务
    asyncio.create_task(task.run())
    
    return task_id

async def watch_progress(task_id: str) -> Iterator[Dict]:
    """监听分析进度"""
    while True:
        task = analysis_tasks.get(task_id)
        if not task:
            break
        
        yield {
            "task_id": task_id,
            "progress": task.progress,
            "step": task.step,
            "status": task.status
        }
        
        if task.status in ["completed", "failed"]:
            break
        
        await asyncio.sleep(0.5)
```

#### 1. `/api/agent/chat` - 对话接口（SSE）

```python
@app.get("/api/agent/chat")
async def chat(message: str, session_id: str):
    """对话接口（SSE）"""
    
    async def event_generator():
        agent = get_agent(session_id)
        
        # 检查是否需要分析页面
        page_path = extract_page_from_message(message)
        
        if page_path and not SkillDatabase().check_analyzed(page_path):
            # 启动后台分析
            task_id = await start_analysis(page_path)
            
            # 推送：需要分析
            yield sse_event("need_analysis", {
                "page": page_path,
                "task_id": task_id
            })
            
            # 推送分析进度
            async for progress in watch_progress(task_id):
                yield sse_event("analysis_progress", progress)
            
            # 推送：分析完成
            task = analysis_tasks[task_id]
            if task.status == "completed":
                yield sse_event("analysis_complete", {
                    "page": page_path,
                    "skills": [s.to_dict() for s in task.result]
                })
        
        # Agent 对话
        async for event in agent.chat(message):
            yield sse_event(event.type, event.data)
    
    return StreamingResponse(
        event_generator(),
        media_type="text/event-stream"
    )
```
